Bài 1.Viết function truyền vào 1 mảng. Kiểm tra mảng đó có phải là mảng tăng dần hay không?
Ví dụ: checkAscArr(1,2,3,4) => true // checkAscArr(1,2,4,3) => false 

            
                let numbers = [1, 2, 4, 5];
                let checkAscArray = numbers.every(function (item, index, arr) {
                    if (index == 0) {
                        return true
                    } else {
                        return item > arr[index - 1];
                    }
                });
                 
                console.log(checkAscArray)

                    let minNum = arr[0]
                    for (let i=0 ; i < arr.length; i++) {
                        if ( arr[i] < minNum ) {
                            minNum = arr[i]
                        }
                    }
                    console.log(minNum)
                    return minNum
                }
                let numbers6 = [1,-2,8,-9]
                findMinNumArr(numbers6)

Bài 7.Viết function truyền vào 1 mảng, kết quả trả về là 1 mảng mới với các số là số dư tương ứng khi chia mảng cũ cho 2
Ví dụ : [4,2,5,6,2,7] => [0,0,1,0,0,1] 

            
                function RemainderArr(arr) {
                    for (let i=0 ; i < arr.length; i++) {
                        let newNum = arr[i]%2
                        arr[i]=newNum
                    }
                    console.log(arr)
                    return arr
                }
                let numbers7 = [4,2,5,6,2,7]
                RemainderArr(numbers7)
